# NS-BRKGA Ablation Study — Stage 1: Vanilla NS-BRKGA
#
# Single population, fixed elite size (min == max), no diversity,
# no exchange, no path relinking, no shaking, no reset.
#
# The synthetic parameter "elites_percentage" is mapped by the ablation
# runner to both --min-elites-percentage and --max-elites-percentage.

population_size_factor      "--population-size-factor "      i   (25, 125)
elites_percentage           "--elites-percentage "           r   (0.10, 0.50)
mutation_probability        "--mutation-probability "        r   (0.00, 1.00)
mutation_distribution       "--mutation-distribution "       r   (1.00, 100.00)
num_total_parents           "--num-total-parents "           i   (2, 5)
num_elite_parents           "--num-elite-parents "           i   (1, 5)
# CONSTANT=0, CUBIC=1, EXPONENTIAL=2, LINEAR=3, LOGINVERSE=4, QUADRATIC=5, SQRT=6
bias_type                "--bias-type "               c       (0, 1, 2, 3, 4, 5, 6)
# ROULETTE=0, GEOMETRIC=1
crossover_type           "--crossover-type "          c       (0, 1)
# Fixed: diversity_type=0 (NONE), num_populations=1, all intervals=0
diversity_type           "--diversity-type "          c       (0)
num_populations          "--num-populations "         c       (1)
exchange_interval        "--exchange-interval "       c       (0)
num_exchange_individuals "--num-exchange-individuals " c      (1)
pr_interval              "--pr-interval "             c       (0)
shake_interval           "--shake-interval "          c       (0)
reset_interval           "--reset-interval "          c       (0)

[global]
digits = 2

[forbidden]
(num_elite_parents > num_total_parents)
(num_elite_parents > (population_size_factor * 4) * elites_percentage)
